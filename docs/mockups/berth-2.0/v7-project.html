export function groupSessionsByCwd(sessions) {
  const pinned = sessions.filter(s => s.pinned);
  const rest = sessions.filter(s => !s.pinned);
  const byCwd = new Map();
  for (const s of rest) {
    (byCwd.get(s.cwd) ?? byCwd.set(s.cwd, []).get(s.cwd)).push(s);
  }
  return { pinned, groups: [...byCwd].sort(byRecent) };
}
